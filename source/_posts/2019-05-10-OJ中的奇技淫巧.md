# OJ中的奇技淫巧

## 实用异或进行swap

```c++
void swap(int& a,int& b){
    a=a^b;
    b=a^b;
    a=a^b;
}
```

## 单行多字符串处理

```c++
#include <bits/stdc++.h>
using namespace std;
int main(){
    string s;
    getline(cin,s);
    stringstream ss;
    ss<<s;
    while(ss>>s){
        cout<<s<<endl;
    }
    return 0;
}
```

## while和scanf

```c++
//读到文件的结尾，程序自动结束
while( ( scanf( " %d " ,&a) ) != -1 )
while( ( scanf( " %d " ,&a) ) != EOF)
while( ( scanf( " %d " ,&a) ) == 1 )
while( ~( scanf( " %d " ,&a) ) )
//读到一个 0 时，程序结束
while( scanf( " %d " ,&a) ,a)
//读到多个0时，程序结束
while( scanf( " %d%d%d " ,&a,&b,&c),a+b+c ) //a,b,c非负
while( scanf( " %d%d%d " ,&a,&b,&c),a|b|c )
```
